#include <stdio.h>
struct h_m_s{
    int hrs;
    int mins;
    int secs;
};
long hms_to_secs(int hrs, int mins, int secs){
    return((long)(hrs*3600L)+(long)(mins*60L)+((long)(secs)));
}
